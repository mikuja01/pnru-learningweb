โครงสร้างของการเขียน User interface

ใน Flutter เกือบทุกอย่างจะเป็น Widget โดย Widget คือ ส่วนต่างๆที่ประกอบกันเป็นส่วนที่แสดงผลบนหน้าจอ เช่น ปุ่ม ข้อความ เมื่อเราต้องการสร้างอะไรที่เกี่ยวข้องกับการแสดงผลบนหน้าจอ โดยจะมีลักษณะซ้อนกันเหมือนกิ่งไม้(Widget Tree)

Widget ต่างๆจะออกแบบให้มีความสัมพันธ์แบบพ่อแม่ลูก เช่น Widget ที่อยู่ภายใน Widget หนึ่งที่ใหญ่กว่าจะเป็นลูก child: หรือ หลาย Widget ย่อยในระดับเดียวกันอยู่ใน Widget ใหญ่ก็จะเป็นเหล่าเด็กๆ children:<Widget>[], หากยังไม่เข้าใจสามารถดูรูปด้านล่างได้เลยนะครับ

จากแผนผัง Widget Tree จะสังเกตได้ว่าจะมีโครงสร้างใหญ่ไปย่อยโดยที่เรียงลำดับได้เป็น(ขอนับแค่ส่วนเนื้อหานะครับ) MyApp>MaterialApp>MyHomePage>Scaffold>Center>Column>Text

หากยังงงอยู่ผมจะแสดง ภาพแบบนี้เลยนะครับ

ขั้นแรกเมื่อเราเปิดแอปขึ้นมา ฟังก์ชัน main() จะทำงานในทันทีโดยที่สั่ง Run MyApp() ซึ่งเป็น WidgetStateless ทำให้ MaterialApp() กำหนดธีมและแสดงหน้า MyHomePage() เป็น WidgetStateful มี Scaffold()ที่ภายในมี Widget กำหนดจุดกึ่งกลาง Center() มี Widget ลูกเป็น Column() เป็น Widget กำหนดรูปแบบ Widget ย่อยให้เรียงกันเป็นคอลัมน์ ซึ่งภายในคอลัมน์ก็จะมีเด็กๆเป็น Text() 2 Widget

ใน WidgetStateful แสดง Widget Scaffold()

และใน Scaffold()มี Widget กำหนดจุดกึ่งกลาง Center() มี Widget ลูกเป็น Column() เป็น Widget กำหนดรูปแบบ Widget ย่อยให้เรียงกันเป็นคอลัมน์ ซึ่งภายในคอลัมน์ก็จะมีเด็กๆเป็น Text() 2 Widget

ศึกษาโครงสร้าง Widget เพิ่มเติม

StatlessWidget และ StatefulWidget

Widget แบ่งเป็น 2 ประเภท คือ Stateless Widget กับ Stateful Widget ก่อนอื่นเข้าใจคำว่า State ก่อน State คือ สถานะของ object หนึ่งๆ หรือก็คือค่าของตัวแปรใน object ในเวลานึงๆ

StatelessWidget

คือ Widget ที่รับค่ามา แล้ว build layout แสดงผลอย่างเดียว ไม่มีเปลี่ยนค่า ไม่มีการ rebuild ด้วยตัวมันเอง เลยไม่จำเป็นต้องมี State เช่น Text , Row, Column พวกนี้รับค่ามา ก็แค่แสดงผลเท่านั้น ไม่มี interaction ใดๆ

โครงสร้างก็จะง่ายๆ คือ extends StatelessWidget แล้วก็ override method build() ซึ่งจะ return Widget กลับไป

หรือจะให้มองง่ายๆคือ StatefulWidget เป็นหน้ากระดาษ 1 หน้าในสมุดที่เราตีตารางตีเส้นขึ้นมาเอง สามารถเขียนและลบข้อความในนั้นได้แต่จะแก้เส้นและตารางในหน้านั้นไม่ได้

class MyHomePage extends StatelessWidget {                  //เป็นเหมือนหน้ากระดาษสมุดที่ชื่อ MyHomepage
  MyHomePage({Key key, this.title}) : super(key: key);      //รับค่าตัวแปร title เข้ามา
  String title;                                             //กำหนดตัวแปร title ขึ้นมาเพื่อรับค่าตัวแปรจากภายนอก
     int _counter = 0;                                      //กำหนดตัวแปรจำนวนเต็มเป็น _counter มีค่าเป็น 0
  void _incrementCounter() {                                //สร้างฟังก์ชัน _incrementCounter
    setState(() {
      _counter++;                                           //เพิ่มค่าจากเดิม +1
    });
  }
  @override
  Widget build(BuildContext context) {                      //สร้าง layout
    return Scaffold(                                        //return widget กลับมาเป็น scaffold
      appBar: AppBar(                                       //MaterialApp ส่วนของ appbar แสดงแถบ AppBar
        title: Text('ตัวอย่างของ Stateless Widget'),
            ),
      body: Center(                                         //Material ส่วน body กำหนด wiget ภายในเริ่มที่กึ่งกลาง
        child: Column(                                      //กำหนดการเรียง widget ภายในเรียงจากบนลงล่าง
          mainAxisAlignment: MainAxisAlignment.center,      //กำหนด widget ใน column ให้อยู่ตรงกลาง
          children: <Widget>[                               //พื้นที่ใส่ widget ลูกๆ
            Text('$_counter'),                              //แสดง ข้อความจากตัวแปร _counter
          ],
        ),
      ),
      floatingActionButton: FloatingActionButton(           //Material ส่วน floatingactionbutton สร้างปุ่ม floating
        onPressed: _incrementCounter,                       //เมื่อมีการกดปุ่มนี้จะเรียกใช้ฟังก์ชัน _incrementCounter
        child: Icon(Icons.add),                             //กำหนดสํญลักษณ์ในปุ่มเป็นรูป add
      ),
    );
  }
}

โดยสามารถทดลองนำโค้ดนี้ไปทดลอง Run และใช้งานดูได้เลยครับเราจะเห็นในสิ่งที่ StatelessWidget นั้นทำไม่ได้

StatefulWidget

Stateful Widget ก็คือ StatelessWidget ที่อัพเกรดนั่นเอง มันสามารถ build layout แล้วยัง rebuild ได้ เมื่อมีค่าข้อมูลเปลี่ยนแปลง เช่น TextField , CheckBox() หรือการสร้างหน้าที่เปลี่ยนแปลง Widget ได้เรื่อยๆ(Dynamic)

class MyHomePage extends StatefulWidget{                    //เป็นเหมือนสมุดชื่อ MyHomepage
  MyHomePage({Key key, this.title}) : super(key: key);      //รับค่าตัวแปร title เข้ามา
  String title;                                             //กำหนดตัวแปร title ขึ้นมาเพื่อรับค่าตัวแปรจากภายนอก
  @override 
  _MyHomePageState createState() => _MyHomePageState();     //ใส่กระดาษลงในสมุด
}
class _MyHomePageState extends State<MyHomePage> {          //เป็นหน้ากระดาษที่จะถูกนำมาใส่ในสมุด
          int _counter = 0;                                 //กำหนดตัวแปรจำนวนเต็มเป็น _counter มีค่าเป็น 0
  void _incrementCounter() {                                //สร้างฟังก์ชัน _incrementCounter
    setState(() {                                           //สร้างหน้ากระดาษใหม่ใส่แทนที่ของเดิมในสมุดแล้วฉีกหน้าเก่าทิ้ง
      _counter++;                                           //เพิ่มค่าจากเดิม +1
    });
  }
  @override
  Widget build(BuildContext context) {                      //สร้าง layout
    return Scaffold(                                        //return widget กลับมาเป็น scaffold
      appBar: AppBar(                                       //MaterialApp ส่วนของ appbar แสดงแถบ AppBar
        title: Text('ตัวอย่างของ Stateless Widget'),
            ),
      body: Center(                                         //Material ส่วน body กำหนด wiget ภายในเริ่มที่กึ่งกลาง
        child: Column(                                      //กำหนดการเรียง widget ภายในเรียงจากบนลงล่าง
          mainAxisAlignment: MainAxisAlignment.center,      //กำหนด widget ใน column ให้อยู่ตรงกลาง
          children: <Widget>[                               //พื้นที่ใส่ widget ลูกๆ
            Text('$_counter'),                              //แสดง ข้อความจากตัวแปร _counter
          ],
        ),
      ),
      floatingActionButton: FloatingActionButton(           //Material ส่วน floatingactionbutton สร้างปุ่ม floating
        onPressed: _incrementCounter,                       //เมื่อมีการกดปุ่มนี้จะเรียกใช้ฟังก์ชัน _incrementCounter
        child: Icon(Icons.add),                             //กำหนดสํญลักษณ์ในปุ่มเป็นรูป add
      ),
    );
  }
}

รายละเอียดของ Widget ต่างๆ

Widget ต่างๆที่เป็นองค์ประองค์กอบใน Widget Tree มีส่วนสำคัญในการแสดง UI ไม่ว่างจะเป็นการหุ้ม Widget อื่นๆอยู่ในกล่องขนาดใหญ่ Container(), หรือการแสดงข้อความ Text(), เป็นต้น

Container

มีคุณสมบัติเหมือนกับตู้ที่ใช้บรรจุ Widget ต่างๆ ซึ่งไม่มีข้อจำกัดว่าจะเล็กหรือใหญ่ เป็นสี่เหลี่ยมหรือวงกลม สามารถที่จะตกแต่ง widget นี้ได้ตามชอบ

ตัวอย่างการนำ Container() มาหุ้มเป็น Widget ใหญ่

StreamBuilder

หากสังเกตการทำงานของแอปนับเลขที่เราได้จากการสร้างโปรเจคในตอนแรกนั้นจะพบว่า เมื่อกดปุ่มจะมีการเรียกใช้ฟังก์ชัน _incrementCounter ที่จะบวกเลขของตัวแปร _counter เพิ่ม 1 ค่า ซึ่งจะแสดงผลตัวเลขที่หน้าแอปตามจำนวนการกดปุ่มที่เพิ่มขึ้นเรื่อยๆ โดยทุกครั้งที่กดปุ่มจะมีการสร้าง Widget ใหม่อยู่ตลอด จากคุณสมบัติของ StatefulWidget แต่ว่าการทำเช่นนั้นจะเป็นการใช้งานที่เกินความจำเป็น ยิ่งกรณีที่มีโครงสร้าง Widget ซ้อนกันหลายชั้นมากๆ ถ้ามีการเปลี่ยนแปลงข้อมูลแค่นิดเดียวก็จะต้องสร้าง Widget ใหม่ทั้งหมดโดยใช่เหตุ

ดังนั้นการเขียนแอพแบบนี้จะทำให้ประสิทธิภาพไม่ดี ถ้ามีข้อมูลแปลี่ยนเป็นบางตัว เราควรจะเรนเดอร์ใหม่เฉพาะวิดเจ็ตที่จำเป็นน่าจะดีกว่า จึงมี Widget พิเศษที่ชื่อว่า StreamBuilder

ศึกษาการใช้ FutureBuilder และ StreamBuilder

วิธีการใช้งาน StreamBuilder() คือ Widget ตัวไหนที่สามารถเรนเดอร์ใหม่ได้เฉพาะส่วน ให้เอา StreamBuilder ครอบ Widget นั้นลงไปแบบนี้

          StreamBuilder(
            stream:_stream(),
            builder: (BuildContext context, AsyncSnapshot snapshot) {
              return Text(
                'เลขที่ออก $_counter',
                style: TextStyle(fontSize: 50),
              );
            },
          )

ซึ่งคุณสมบัติของ Widget สามารถทำให้สร้างแอปนับเลขได้ใน StatelessWidget

แต่การใช้ Widget นี้จะต้องกำหนด ฟังก์ชันในการ stream เช่น

            Stream<int> _stream() async* {
              while (true) {
                await Future.delayed(Duration(seconds: 1));
                yield c++;
              }
            }

คือ การอัพเดท stream ในทุกๆ1วินาที โดยอ้างอิงตามตัวแปร c
หรือสามารถใช้ฟังก์ชัน stream จาก Package ก็สามารถทำได้
